Test Case 1:


// Define the Laplace variable
s = %s;

// Define the transfer function
g = syslin('c', 1 / (s^2 + 2 * s + 1));

// Convert the system to state-space representation
ss_g = tf2ss(g);

// Define time vector for simulation
t = 0:0.01:10;

// Define a step input
u = ones(1, length(t));

// Compute the step response using the defined transfer function and time vector
y = _step_(u, t, ss_g);

// Plot step response
plot(t, y);
title('Step Response of a PT2 Transfer Function');
xlabel('Time [s]');
ylabel('y1');
